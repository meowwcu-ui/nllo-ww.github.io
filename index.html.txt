<script>
  (function() {
    // Элементы
    var page1 = document.getElementById('page1');
    var page2 = document.getElementById('page2');
    var page3 = document.getElementById('page3');
    var toPage2Btn = document.getElementById('toPage2');
    var btnYes = document.getElementById('btnYes');
    var btnNo = document.getElementById('btnNo');
    var yesPopup = document.getElementById('yesPopup');
    var noPopup = document.getElementById('noPopup');
    var closeNoPopup = document.getElementById('closeNoPopup');
    var nextLink = document.getElementById('nextLink');
    var backToPage2From3 = document.getElementById('backToPage2From3');
    var bgMusic = document.getElementById('bgMusic');
    var loveVideo = document.getElementById('loveVideo');
    var telegramMessage = document.getElementById('telegramMessage');
    var heartReaction = document.getElementById('heartReaction');

    // Фон "love you"
    var lovePattern = document.getElementById('lovePattern');
    for (var i = 0; i < 30; i++) {
      var span = document.createElement('span');
      span.textContent = 'love you';
      span.style.left = (i % 6) * 20 + '%';
      span.style.top = Math.floor(i / 6) * 25 + '%';
      span.style.fontSize = (20 + Math.random() * 20) + 'px';
      span.style.transform = 'rotate(' + (-15 + Math.random() * 30) + 'deg)';
      lovePattern.appendChild(span);
    }

    // Статичные сердечки
    var staticHearts = document.getElementById('staticHearts');
    for (var i = 0; i < 25; i++) {
      var span = document.createElement('span');
      span.textContent = '❤️';
      span.style.left = Math.random() * 100 + '%';
      span.style.top = Math.random() * 100 + '%';
      span.style.fontSize = (25 + Math.random() * 30) + 'px';
      staticHearts.appendChild(span);
    }

    // Переходы
    toPage2Btn.addEventListener('click', function() {
      page1.classList.remove('active');
      page2.classList.add('active');
    });

    // Логика страницы 2
    btnYes.addEventListener('click', function() {
      yesPopup.classList.remove('hidden');
      nextLink.classList.remove('hidden');
      if (bgMusic) {
        bgMusic.play().catch(function() {});
      }
    });

    btnNo.addEventListener('click', function() {
      noPopup.classList.remove('hidden');
    });

    closeNoPopup.addEventListener('click', function() {
      noPopup.classList.add('hidden');
    });

    yesPopup.addEventListener('click', function(e) {
      if (e.target === yesPopup) yesPopup.classList.add('hidden');
    });

    noPopup.addEventListener('click', function(e) {
      if (e.target === noPopup) noPopup.classList.add('hidden');
    });

    nextLink.addEventListener('click', function() {
      page2.classList.remove('active');
      page3.classList.add('active');
      if (loveVideo) {
        loveVideo.play().catch(function() {});
      }
      if (bgMusic) {
        bgMusic.pause();
      }
    });

    // Страница 3
    backToPage2From3.addEventListener('click', function() {
      page3.classList.remove('active');
      page2.classList.add('active');
      if (loveVideo) loveVideo.pause();
    });

    // Двойное нажатие
    var tapCount = 0;
    var tapTimer = null;
    telegramMessage.addEventListener('click', function() {
      if (!heartReaction.classList.contains('hidden')) return;
      tapCount++;
      if (tapCount === 1) {
        tapTimer = setTimeout(function() {
          tapCount = 0;
        }, 300);
      } else if (tapCount === 2) {
        clearTimeout(tapTimer);
        tapCount = 0;
        heartReaction.classList.remove('hidden');
      }
    });

  })();
</script>
